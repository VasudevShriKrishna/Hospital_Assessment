import folium
from django.shortcuts import render
from .data_loader import load_hospitals
from .utils import find_nearest_hospital


def map_view(request):
    selected_lat = request.GET.get("lat")
    selected_lon = request.GET.get("lon")

    m = folium.Map(location=[39.5, -98.35], zoom_start=4)

    hospitals = load_hospitals()

    # Add all hospitals as small blue markers
    for hospital in hospitals:
        coords = hospital["geometry"]["coordinates"]
        h_lon = coords[0]
        h_lat = coords[1]
        name = hospital["properties"]["NAME"]

        folium.CircleMarker(
            [h_lat, h_lon],
            radius=3,
            color="blue",
            fill=True,
            fill_opacity=0.7,
            popup=name
        ).add_to(m)

    # If user clicked somewhere
    if selected_lat and selected_lon:
        lat = float(selected_lat)
        lon = float(selected_lon)

        nearest, distance = find_nearest_hospital(lat, lon, hospitals)

        props = nearest["properties"]

        popup_html = f"""
        <b>Nearest Hospital</b><br>
        Name: {props['NAME']}<br>
        Address: {props['ADDRESS']}<br>
        City: {props['CITY']}, {props['STATE']}<br>
        Beds: {props['BEDS']}<br>
        Distance: {distance} km
        """

        # Red marker at selected location
        folium.Marker(
            [lat, lon],
            popup=popup_html,
            icon=folium.Icon(color="red", icon="info-sign")
        ).add_to(m)

    return render(request, "core/map.html", {"map": m._repr_html_()})

<script>
let clickCount = 0;
let lastClickTime = 0;

setTimeout(function() {

    let map = window._leaflet_map;

    map.on("click", function(e) {

        let lat = e.latlng.lat;
        let lon = e.latlng.lng;

        document.getElementById("coords").innerText =
            "Lat: " + lat.toFixed(6) +
            " | Lon: " + lon.toFixed(6);

        let now = new Date().getTime();

        if (now - lastClickTime < 700) {
            clickCount++;
        } else {
            clickCount = 1;
        }

        lastClickTime = now;

        if (clickCount === 3) {

            fetch(`/nearest/?lat=${lat}&lon=${lon}`)
                .then(response => response.json())
                .then(data => {

                    document.getElementById("result").innerHTML =
                        "<b>Nearest Hospital:</b> " + data.hospital +
                        "<br>Distance: " + data.distance_km + " km" +
                        "<br>Earthquake Risk: " + data.earthquake_risk +
                        "<br>Temperature: " + data.temperature_c + " °C" +
                        "<br>Wind Speed: " + data.windspeed_kmh + " km/h";
                });

            clickCount = 0;
        }
    });

}, 1000);
</script>

</body>
